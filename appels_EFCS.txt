## Recherche de films - Cas 1 : Sans critère avec pagination (demandez la page 1)

query {
  filmsPaginated (page:1){
    data {
      id,
      title,
      description,
      release_year,
      length,
    }
    paginatorInfo {
      currentPage
      lastPage,
    }
  }
}

## Recherche de films - Cas 2 : Avec le mot-clé "AL" (sur titre ou descritpion) avec pagination (demandez la page 1)

query {
  filmsFiltreByTitle (title: "%AL%", page:1){
    data {
      id,
      title,
      description,
      release_year,
      length,
    }
    paginatorInfo {
      currentPage
      lastPage,
    }
  }
}

## Recherche de films - Cas 3 : Avec le mot-clé "AL" (sur titre ou descritpion), minimum année 2006, entre 120 et 150 minutes avec pagination (demandez la page 1)

query {
  filmsFiltreByDescription (description: "%AL%", 
    release_year : 2006, length_min : 120, length_max : 150, 
    page:1){
    data {
      id,
      title,
      description,
      release_year,
      length,
    }
    paginatorInfo {
      currentPage
      lastPage,
    }
  }
}
